155 void Tonav::performUpdate() {
156  ROS_INFO("PERFORMING UPDATE");
157 
158  double accelerometer_time = accel_buffer_.front().getTime();
159  double gyroscope_time = gyro_buffer_.front().getTime();
160 
161  ImuItem accelerometer_at_image_time;
162  ImuItem gyroscope_at_image_time;
163 
164  while (accelerometer_time < last_image_capture_time_ || gyroscope_time < last_image_capture_time_) {
165  // Propagate IMU
166  if (accelerometer_time < gyroscope_time) {
167  std::list<ImuItem>::iterator earlier_it = std::begin(accel_buffer_);
168  std::list<ImuItem>::iterator later_it = ++std::begin(accel_buffer_);
169 
170  if (earlier_it->getTime() <= last_image_capture_time_ && later_it->getTime() > last_image_capture_time_) {
171  accelerometer_at_image_time = interpolate(last_image_capture_time_, *earlier_it, *later_it);
172  }
173 
174  ImuItem accelerometer = interpolate(gyroscope_time, *earlier_it, *later_it);
175  filter_.stepInertial(gyroscope_time - last_update_time_, accelerometer, gyro_buffer_.front());
176  last_update_time_ = gyroscope_time;
177  accel_buffer_.pop_front();
178  accelerometer_time = accel_buffer_.front().getTime();
179  } else {
180  std::list<ImuItem>::iterator earlier_it = std::begin(gyro_buffer_);
181  std::list<ImuItem>::iterator later_it = ++std::begin(gyro_buffer_);
182 
183  if (earlier_it->getTime() <= last_image_capture_time_ && later_it->getTime() > last_image_capture_time_) {
184  gyroscope_at_image_time = interpolate(last_image_capture_time_, *earlier_it, *later_it);
185  }
186 
187  ImuItem gyroscope = interpolate(accelerometer_time, *earlier_it, *later_it);
188  filter_.stepInertial(accelerometer_time - last_update_time_, accel_buffer_.front(), gyroscope);
189  last_update_time_ = accelerometer_time;
190  gyro_buffer_.pop_front();
191  gyroscope_time = gyro_buffer_.front().getTime();
192  }
193  }
194 
195  // Update image
196  filter_.stepCamera(last_image_capture_time_ - last_update_time_, accelerometer_at_image_time, gyroscope_at_image_time, last_image_);
197  last_update_time_ = last_image_capture_time_;
198  last_image_capture_time_ = NAN;
199 }
